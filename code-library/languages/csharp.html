C# Programming

Key C# Features

Console.WriteLine()

Outputs text to the console

Console.WriteLine(value);

String.Format()

Formats strings with placeholders

String.Format(format, args);

List<T>

Generic collection of strongly typed objects

List<type> name = new List<type>();

LINQ

Language Integrated Query for data operations

from item in collection where condition select item;

C# Code Examples

Basic Syntax

// Variables and types
string name = "Alice";
int age = 25;
bool isStudent = true;

// Conditional statement
if (age >= 18)
{
    Console.WriteLine($"{name} is an adult");
}
else
{
    Console.WriteLine($"{name} is a minor");
}

// Array initialization
int[] numbers = { 1, 2, 3, 4, 5 };

Object-Oriented Programming

// Class definition
public class Person
{
    // Properties
    public string Name { get; set; }
    public int Age { get; set; }

    // Constructor
    public Person(string name, int age)
    {
        Name = name;
        Age = age;
    }

    // Method
    public string Introduce()
    {
        return $"My name is {Name} and I'm {Age} years old.";
    }
}

// Usage
var person = new Person("Bob", 30);
Console.WriteLine(person.Introduce());

Advanced Features

// Inheritance
public class Student : Person
{
    public string StudentId { get; set; }
    
    public Student(string name, int age, string studentId) 
        : base(name, age)
    {
        StudentId = studentId;
    }
}

// Interface
public interface IShape
{
    double Area();
}

// LINQ query
var numbers = new List<int> { 1, 2, 3, 4, 5 };
var evenNumbers = from n in numbers 
                  where n % 2 == 0 
                  select n;
